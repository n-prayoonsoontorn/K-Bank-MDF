# Function to sum up values in a specific column
def sum_column(records, column_name):
    # Find the index of the column
    column_index = None
    for i, column in enumerate(records[0]):
        if column.strip('"') == column_name:
            column_index = i
            break
    
    if column_index is None:
        return None  # Column not found
    
    # Sum up the values in the column
    total = 0
    for record in records[1:]:
        value = float(record[column_index].strip('"'))
        total += value
    
    return total
